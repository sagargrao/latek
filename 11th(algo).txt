\section *{Floyd's Algorithm}

Algorithm to find solution to all-pairs shortest-path problem.

\begin{algorithm}[H]

\KwIn{The weight matrix $W$ of a graph having vertices $[1 \ldots n]$}

\KwOut{The distance matrix $D$ of the shortest path length between every pair of vertices $[1 \ldots n]$}

$D \gets W$

\For{$k \gets 1$ to $n$}
{
    \For{$i \gets 1$ to $n$}
    {
        \For{$j \gets 1$ to $n$}
        {
            $D[i,j] \gets \min(D[i,j], D[i,k] + D[k,j])$
        }
    }
}
